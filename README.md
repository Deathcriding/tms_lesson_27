```python
#!/usr/bin/env python3

# =====================================================
# УСЛОВНЫЕ ОПЕРАТОРЫ
# =====================================================

# Простое условие
age = 18
if age >= 18:
    print("Вы совершеннолетний")

    print("Вы несовершеннолетний")

age = 25
has_license = True

if age >= 18 and has_license:
    print("Можно водить машину")
else:
    print("Нельзя водить машину")

if age < 18 or age > 60:
    print("Льготная категория")

if not has_license:
    print("Нет прав")
```

# ЭТАП 3: ЦИКЛЫ

## Демонстрация 4: Циклы for и while

**Создайте файл `demo_loops.py`:**

```python
#!/usr/bin/env python3

# =====================================================
# ЦИКЛ FOR
# =====================================================

print("=== Цикл по списку ===")
fruits = ["яблоко", "банан", "апельсин", "груша"]
for fruit in fruits:
    print(f"Фрукт: {fruit}")

print("\n=== range(5) ===")
for i in range(5):
    print(f"i = {i}")

print("\n=== range(1, 10, 2) ===")
for i in range(1, 10, 2):
    print(f"Нечетное: {i}")

print("\n=== range(10, 0, -1) ===")
for i in range(10, 0, -1):
    print(f"Обратный отсчет: {i}")

# =====================================================
# ЦИКЛ ПО СЛОВАРЮ
# =====================================================

person = {"name": "Анна", "age": 22, "city": "Казань"}

print("\n=== Перебор словаря ===")
for key, value in person.items():
    print(f"{key}: {value}")

# =====================================================
# enumerate()
# =====================================================

colors = ["красный", "синий", "зеленый"]

print("\n=== С индексом ===")
for i, color in enumerate(colors, 1):
    print(f"{i}. {color}")

# =====================================================
# zip()
# =====================================================

names = ["Анна", "Петр", "Мария"]
ages = [22, 30, 25]

print("\n=== Объединение списков ===")
for name, age in zip(names, ages):
    print(f"{name} - {age} лет")

# =====================================================
# ЦИКЛ WHILE
# =====================================================

print("\n=== Цикл while ===")
count = 1
while count <= 5:
    print(f"Итерация {count}")
    count += 1

# =====================================================
# BREAK, CONTINUE
# =====================================================

print("\n=== break ===")
for i in range(10):
    if i == 5:
        break
    print(f"i = {i}")

print("\n=== continue ===")
for i in range(10):
    if i % 2 == 0:
        continue
    print(f"Нечетное: {i}")

# =====================================================
# ВЛОЖЕННЫЕ ЦИКЛЫ
# =====================================================

print("\n=== Таблица умножения ===")
for i in range(1, 6):
    for j in range(1, 6):
        print(f"{i*j:2d}", end=" ")
    print()

# =====================================================
# LIST COMPREHENSION
# =====================================================

print("\n=== Квадраты чисел ===")
squares = [x**2 for x in range(1, 11)]
print(f"Квадраты: {squares}")

print("\n=== Четные числа ===")
evens = [x for x in range(1, 21) if x % 2 == 0]
print(f"Четные: {evens}")
```

```python
#!/usr/bin/env python3

# =====================================================
# ПРОСТАЯ ФУНКЦИЯ
# =====================================================

def greet(name):
    """Простая функция приветствия"""
    print(f"Привет, {name}!")

# =====================================================
# ФУНКЦИЯ С ВОЗВРАТОМ
# =====================================================

def add(a, b):
    return a + b

# =====================================================
# ФУНКЦИЯ С ПАРАМЕТРОМ ПО УМОЛЧАНИЮ
# =====================================================

def calculate(a, b, operation="+"):
    if operation == "+":
        return a + b
    elif operation == "-":
        return a - b
    elif operation == "*":
        return a * b
    elif operation == "/":
        if b == 0:
            return "Ошибка: деление на ноль"
        return a / b
    return "Неизвестная операция"

# =====================================================
# *args И **kwargs
# =====================================================

def sum_all(*args):
    return sum(args)

def print_info(**kwargs):
    for key, value in kwargs.items():
        print(f"{key}: {value}")

# =====================================================
# ЛЯМБДА-ФУНКЦИИ
# =====================================================

square = lambda x: x ** 2
double = lambda x: x * 2

# =====================================================
# ВЫЗОВ
# =====================================================

greet("Мария")
print(f"15 + 30 = {add(15, 30)}")
print(f"10 / 5 = {calculate(10, 5, '/')}")
print(f"Сумма: {sum_all(1, 2, 3, 4, 5)}")
print_info(name="Анна", age=22, city="Казань")
print(f"Квадрат 5: {square(5)}")
```

# ЭТАП 5: ФАЙЛЫ

## Демонстрация 6: Работа с файлами

**Создайте файл `demo_files.py`:**

```python
#!/usr/bin/env python3

import os
import json
import csv

# =====================================================
# ЗАПИСЬ В ФАЙЛ
# =====================================================

# Запись строк
with open("test.txt", "w") as f:
    f.write("Первая строка\n")
    f.write("Вторая строка\n")
    f.write("Третья строка\n")

print("Файл test.txt создан")

# =====================================================
# ЧТЕНИЕ ИЗ ФАЙЛА
# =====================================================

with open("test.txt", "r") as f:
    content = f.read()
    print("\n=== Содержимое файла ===")
    print(content)

# Чтение построчно
with open("test.txt", "r") as f:
    print("\n=== Построчное чтение ===")
    for line in f:
        print(f"Строка: {line.strip()}")

# =====================================================
# JSON
# =====================================================

data = {
    "name": "Анна",
    "age": 22,
    "city": "Казань",
    "hobbies": ["чтение", "спорт"]
}

# Запись JSON
with open("data.json", "w") as f:
    json.dump(data, f, ensure_ascii=False, indent=2)

print("\nJSON файл создан")

# Чтение JSON
with open("data.json", "r") as f:
    loaded_data = json.load(f)
    print(f"Загружено: {loaded_data}")

# =====================================================
# CSV
# =====================================================

# Запись CSV
with open("users.csv", "w", newline='') as f:
    writer = csv.writer(f)
    writer.writerow(["Имя", "Возраст", "Город"])
    writer.writerow(["Анна", 22, "Казань"])
    writer.writerow(["Петр", 30, "Москва"])

print("\nCSV файл создан")

# Чтение CSV
with open("users.csv", "r") as f:
    reader = csv.reader(f)
    print("\n=== CSV данные ===")
    for row in reader:
        print(row)

# =====================================================
# ПРОВЕРКА СУЩЕСТВОВАНИЯ
# =====================================================

if os.path.exists("test.txt"):
    print("\nФайл test.txt существует")
    print(f"Размер: {os.path.getsize('test.txt')} байт")
```
